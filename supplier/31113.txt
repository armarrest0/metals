SELECT COUNT(distinct(g.goods_id)) FROM `ecshop`.`ecs_goods` AS g,`ecshop`.`ecs_supplier_goods_cat` as sgc, `ecshop`.`ecs_virtual_district` as dis, `ecshop`.`ecs_virtual_goods_district` as dig WHERE dis.supplier_id = 12 AND sgc.goods_id=g.goods_id  and dis.goods_id = g.goods_id and dis.district_id = dig.district_id AND is_delete='0' AND sgc.supplier_id='12'  AND extension_code='virtual_good' AND is_real='0'
